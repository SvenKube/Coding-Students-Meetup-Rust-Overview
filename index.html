Pattern Matching

Patterns


let x = 1;

match x {
    1 => println!("one"),
    2 => println!("two"),
    3 => println!("three"),
    _ => println!("anything"),
}

Multiple patterns


let x = 1;

match x {
    1 | 2 => println!("one or two"),
    3 => println!("three"),
    _ => println!("anything"),
}

Destructuring


struct Point {
    x: i32,
    y: i32,
}

let some_point = Point { x: 0, y: 1 };

match some_point {
